// Fibonacci sequence generator
function* fibonacci() {
  let [a, b] = [0, 1];
  while (true) {
    yield a;
    [a, b] = [b, a + b];
  }
}

// Get first 10 numbers
const fib = fibonacci();
const result = [];
for (let i = 0; i < 10; i++) {
  result.push(fib.next().value);
}
console.log(result);
